1. For the given JSON iterate over all for loops (for, for in, for of, forEach)
	(I used JSON from https://raw.githubusercontent.com/Rajavasanthan/jsondata/master/pagenation.json)

	// forEach
	inp.forEach((item) => {
	  console.log(item.name);
	});

	// for
	for (var i = 0; i < inp.length; i++) {
	  console.log(inp[i].name);
	};

	// for in
	for (var i in inp) {
	  console.log(inp[i].name);
	};

	//for of
	for (var iter of inp) {
	  console.log(iter.name);
	};
